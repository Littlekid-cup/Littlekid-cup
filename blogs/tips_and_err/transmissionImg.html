    formdata格式传输图片文件

    一江秋水 1/9/2023 vuetips

    # 通过 element-ui upload 组件自定义上传，以 formdata 格式进行图片传输（注意：请求头设置"Content-Type": "multipart/form-data",）

    # 最终实现的目的是将前端上传的图片文件以 form 格式传给后端，后端将图片保存下来并返回给前端图片的访问地址。

    # 前端（vue）

    upload 组件

    //基于element-upload重新封装upload封装组件
<template>
  <!-- https://blog.csdn.net/yanmuchen/article/details/121614961 -->
  <div class="upload-box">
    <el-upload
      class="upload"
      ref="upload"
      :limit="1"
      :action="uploadUrl"
      :auto-upload="false"
      :on-change="changeFile"
      :show-file-list="false"
    >
      <img
        class="other-img"
        v-if="fileList[0]?.url"
        :src="fileList[0]?.url"
        alt=""
      />
      <i v-else class="el-icon-plus avatar-uploader-icon"></i>
      <i class="clean-image" @click.stop="cleanImage"></i>
    </el-upload>
  </div>
</template>

<script>
export default {
  name: "upload-vue",
  data() {
    return {
      uploadUrl: ``,
      fileList: [],
    };
  },
  mounted() {
    // this.$bus.$on("submitUpload", () => {
    //   this.submit();
    // });
  },
  methods: {
    beforeOtherPictureUpload(file) {
      const isImage = ["image/jpeg", "image/jpg", "image/png"].includes(
        file.raw.type
      );
      const isLt4M = file.size / 1024 / 1024 <= 8;

      if (!isImage) {
        this.$message.error("the type only suport 'jpg' or 'png'!");
        return false;
      }
      if (!isLt4M) {
        this.$message.error("the avatar should be no more than 8MB!");
        return false;
      }
      return isImage && isLt4M;
    },
    changeFile(file) {
      if (!this.beforeOtherPictureUpload(file)) {
        this.cleanImage(); //上传格式失败，uploadFiles里也会有文件
        return;
      }
      let obj = {
        name: file.name,
        url: URL.createObjectURL(file.raw),
        type: file.raw.type,
        raw: file.raw,
      };
      this.fileList.push(obj);
      this.$emit("getFile", this.fileList[0]);
    },
    cleanImage() {
      this.$refs.upload.uploadFiles = [];
      if (this.fileList.length !== 0) {
        this.$emit("cleanFile", this.fileList[0]?.name);
        this.fileList = [];
      }
    },
  },
};
</script>

    index.js

    //secondImg.json 文件 {"baseUrl":"http://127.0.0.1:3300/","imageUrl":["bg.jpg","worlds.png"]}
const express = require("express");
const app = express();
const fs = require("fs");
const path = require("path");
const multer = require("multer");
let objMulter = multer({ dest: "./uploads" });

const firstJson = require("./data/json.json");
const secondImg = require("./data/secondImg.json");

app.use(function (req, res, next) {
  res.header("Access-Control-Allow-Origin", "*");
  res.header("Access-Control-Allow-Methods", "PUT,GET,POST,DELETE,OPTIONS");
  res.header("Access-Control-Allow-Headers", "X-Requested-With");
  res.header("Access-Control-Allow-Headers", "Content-Type");
  next();
});
// app.use(formidable());
app.use(objMulter.any());
app.use(express.static("./uploads")); //托管文件，用于访问

/**
 * 返回数据处理
 * @param {*} jsongData 返回的data具体数据（json）
 * @param {*} success 是否成功（boolean）
 * @returns
 */
function handleData(jsongData, success) {
  let obj = {
    code: "",
    message: "",
    data: null,
  };
  if (success) {
    obj.code = 200;
    obj.message = "success";
    obj.data = JSON.parse(jsongData);
  }
  return JSON.stringify(obj);
}

//上传图片保存到本地
app.post("/second/submitImg", async (req, res) => {
  let secondObj = JSON.parse(secondImg); //读取json转换成对象
  // let temp = Object.entries(secondObj.imageUrl);
  for (let i = 0; i < req.files.length; i++) {
    let oldName = req.files[i].filename; //自动生成的
    let newName = req.files[i].originalname;
    fs.renameSync("./uploads/" + oldName, "./uploads/" + newName);
    secondObj.imageUrl.push(newName);
  }
  //本地json修改
  await fs.writeFile(
    "./data/secondImg.json",
    JSON.stringify(secondObj),
    (err) => {
      if (err) return console.log("失败");
      console.log("成功 ");
    }
  );
  res.send({
    code: 200,
    url: "http://127.0.0.1:3300/",
    name: secondObj.imageUrl,
  });
  // res.send(firstJson);
});

//获取图片返回
app.get("/second/getImg", async (req, res) => {
  // res.send(firstJson);
  await fs.readFile("./data/secondImg.json", "utf-8", (err, dataStr) => {
    res.send(handleData(dataStr, true));
  });
});

app.listen(3300, () => {
  console.log("serve is run in 3300");
});
