    # 关于 webpack 反向代理

    • 用途： 仅开发时解决前端调用接口跨域问题。（后端使用 nginx 配置过则不考虑）
    • 适用前提： 开发时前端发送请求显示跨域。

    # 反向代理 vue.config.js 配置

    //在vue.config.js中的配置
const { defineConfig } = require("@vue/cli-service");
module.exports = defineConfig({
  transpileDependencies: true,
  publicPath: "./",
  pages: {
    index: {
      entry: "src/main.js", // 入口文件
      title: "你的标题",
    },
  },
  //反向代理 baseurl改为空则api路径需带上/api,
  devServer: {
    proxy: {
      //代理多个需要将其配置到前面，（配置到对象后面只有第一个可以用）
      "/is": {
        target: "https://api.other.com/", //
        secure: false, //https 需要开
        changeOrigin: true, //跨域需要开
        // ws: true, //如果要代理 websockets，配置这个参数
        pathRewrite: {
          "^/is": "",
        },
      },
      // 代表将来如果url地址以/api开头，就会自动代理到对应的服务器
      "/api": {
        target: "https://api.snh48.com/", // 最新写法proxy对象下对应的路径都是一个对象，必须用target指定地址.
        secure: false, //https 需要开
        changeOrigin: true, //跨域需要开
        // ws: true, //如果要代理 websockets，配置这个参数
        pathRewrite: {
          "^/api": "",
        },
        //后端设置referer的防伪
        headers: {
          referer: "https://api.snh48.com/",
          // origin: 'https://api.snh48.com/'
        },
      },
    },
  },
});

    # 创建 axios 实例时配置 baseURL 以及其他参数

    // 在request.js创建axios实例
import axios from "axios";

let baseURL = "";
//判断开发环境
if (process.env.NODE_ENV === "development") {
  baseURL = "/api";
} else {
  if (process.env.type === "test") {
    baseURL = "http://api.snh48.com";
  } else {
    baseURL = "https://api.snh48.com";
  }
}

//axios默认设置
axios.defaults.timeout = 6000;
axios.defaults.baseURL = baseURL;
axios.defaults.responseType = "json";
axios.defaults.withCredentials = false;

//axios实例
let request = axios.create({
  baseURL,
  timeout: "8000",
  withCredentials: false, //跨域请求时发送cookies
});
//请求拦截器----在项目中发请求（请求没有发出去）可以做一些事情
request.interceptors.request.use((config) => {
  // config.headers.referer = "https://www.snh48.com";
  return config;
});

//响应拦截器----当服务器手动请求之后，做出响应（相应成功）会执行的
request.interceptors.response.use(
  (res) => {
    //相应成功做的事情
    return res.data;
  },
  (err) => {
    alert(err + "服务器响应数据失败");
  }
);

//这里的代码是暴露一个axios实例
export default request;

export function getMvByTag() {
  return request({
    method: "get",
    url: "/getMvByTag",
  });
